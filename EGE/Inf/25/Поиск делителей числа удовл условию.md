def all_divs(x):
	divs = []
	for d in range(2, int(x**0.5) + 1):
		if x % d == 0:
			divs.append(d)
			if d != x // d:
				divs.append(x//d)
	return sorted(divs)

count = 0
cnum = 650000

while count <= 5:
    if len(all_divs(cnum)) == 6 and len(str(max(all_divs(cnum)) + min(all_divs(cnum)))) == 4:
        print(cnum, (max(all_divs(cnum)) + min(all_divs(cnum))))
        count += 1
        if count == 5:
            break
    cnum += 1
