```python
from itertools import *

alphabet = '012345678'
t = ['00', '11', '22', '33', '44', '55', '66', '77', '88']
count = 0
for i in product(alphabet, repeat=5):
	p = ''.join(i)
	if p[0] != 0 and p.count('5')==1 and all(not i in p for i in t):
		count += 1

print(count)
```
