```python
from ipaddress import *
def determ_bin_str(ip):
	s = ip.split('.')
	string = ''
	for value in s:
		string = string + str(bin(int(value))[2:])
	if string.count('1') % 2 != 0:
		print(string)
network = ip_network('192.168.32.48/255.255.255.240')
for ip in network.hosts():
	determ_bin_str(str(ip))
```
